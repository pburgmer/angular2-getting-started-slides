Komponenten mit Data-Binding 
import { Track } from "./../tracks/track.model";
@Component({
  selector: "music-app",
  template: `
    
Music App

    Music App has {{tracks.length}} tracks to play
  `
})
export class MusicAppComponent {
  public tracks: Track[];
  constructor() {
    this.tracks = [
      new Track("Sound of Silence"), new Track("I Wish"),
      new Track("Over My Shoulder"), new Track("Take Five")
    ];
  }
}
