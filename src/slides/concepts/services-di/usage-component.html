Dependency Injection
  • Verwendung bei Komponenten
    • Injektion als Konstruktor-Parameter
    • Kein @Injectable() notwendig, da @Component verwendet
    @Component({ selector: "release-card" })
export class ReleaseCardComponent {
  constructor(private log: Logger) {}
  likeRelease() { this.log.debug(`Release ${this.release.name} liked`); }
}
