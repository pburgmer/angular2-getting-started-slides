Http MockBackend
  • MockBackend in Test injizieren lassen
  • MockBackend stellt Observable f√ºr Verbindungen bereit
    • Wenn Verbindung angefordert wird, wird Subscriber aufgerufen
    • Subscriber kann Antwort liefern
  • Spy als Subscriber -> Testen wie viele Connections
it('should get tracks',
  async(inject([MockBackend, TrackService], (mockBackend, trackService) => {
    mockBackend.connections.subscribe((connection: MockConnection) => {
      connection.mockRespond(new Response(
        new ResponseOptions({ body: [{ id: 0, name: "Sound of Silence" }]})
      ));
    });

    trackService.getTracks().subscribe(tracks => {
      expect(tracks.length).toBe(1);
      expect(tracks[0].id).toBe(0);
    });
  }))
);
